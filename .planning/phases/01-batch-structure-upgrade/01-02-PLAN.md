         const needsRebuild = sheet.getLastColumn() < 15
           || JSON.stringify(sheet.getRange(1, 1, 1, sheet.getLastColumn()).getValues()[0]) !== JSON.stringify(expectedRecHeader);
         if (needsRebuild) {
           sheet.clear();
           sheet.appendRow(expectedRecHeader);

           sheet.setFrozenRows(1);
         }
